# MDate
```
import MDate from './MDate.js'

new MDate().Format('YYYY-MM-DD hh:mm:ss qq S');//  2016-12-01 14:20:01 04 607
new MDate().Format('YYYY-MM-DD');//  2016-12-01


new MDate('2015-11-01 15:25:58').Format('YYYY-MM-DD hh:mm:ss qq S');//  2015-11-01 15:25:58 04 0
new MDate('2015-11-01 15:25:58').Format('YYYY-MM-DD');//  2015-11-01

```
